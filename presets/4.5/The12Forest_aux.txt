#$ TITLE: The12Forest AUX Mode Assignments
#$ FIRMWARE_VERSION: 4.5
#$ CATEGORY: MODES
#$ STATUS: COMMUNITY
#$ KEYWORDS: aux, modes, arm, channel, map, receiver, elrs, The12Forest
#$ AUTHOR: The12Forest
#$ DESCRIPTION: Receiver channel map and AUX mode assignments used by The12Forest across Betaflight 4.5 builds.
#$ DESCRIPTION: Channel map ATER1234 is used — this matches the default ELRS/ExpressLRS channel order where
#$ DESCRIPTION: channels 1–4 are Aileron, Throttle, Elevator, Rudder followed by AUX channels.
#$ DESCRIPTION: AUX 0 / AUX 1 handle arming: AUX0 arms when CH1 is above 1525, AUX1 provides a secondary arm range.
#$ DESCRIPTION: AUX 2 activates Angle mode at high switch position (above 1800 on CH2).
#$ DESCRIPTION: AUX 3 maps to GPS Rescue (mode 46) on CH3 above 1800.
#$ DESCRIPTION: AUX 4 maps to Turtle / Flip-over-after-crash (mode 13) in the mid position on CH3.
#$ DESCRIPTION: AUX 5 activates Blackbox logging (mode 28) on CH2 above 1250.
#$ DESCRIPTION: AUX 6 is a permanent always-off assignment used to keep a mode slot reserved.
#$ DESCRIPTION: AUX 7–9 assign to user modes 35, 36, 37 on CH4 / CH5 for camera control and VTX pit mode.
#$ DESCRIPTION: RX failsafe overrides set CH4 (rudder/yaw) to 1550 (center) and CH6 to 2000 (activate rescue) on signal loss.
#$ DISCLAIMER: Switch positions and channel numbers are specific to The12Forest's transmitter layout. Review before flying.
#$ DISCUSSION: https://github.com/The12Forest/Betaflight-Presets

defaults nosave


#$ OPTION_GROUP BEGIN: Receiver Channel Map
    #$ OPTION BEGIN (CHECKED): Channel map ATER1234
        map ATER1234
    #$ OPTION END
#$ OPTION_GROUP END


#$ OPTION_GROUP BEGIN: AUX Mode Assignments
    #$ OPTION BEGIN (CHECKED): AUX mode assignments
        aux 0 0 0 1525 2100 0 0
        aux 1 0 1 1300 2100 1 0
        aux 2 1 1 1800 2100 0 0
        aux 3 46 2 1800 2100 0 0
        aux 4 13 2 1350 1650 0 0
        aux 5 28 1 1250 2100 0 0
        aux 6 30 0 900 900 0 1
        aux 7 35 3 1800 2100 0 0
        aux 8 36 4 1425 2100 0 0
        aux 9 37 5 1825 2100 0 0
    #$ OPTION END
    #$ OPTION BEGIN (CHECKED): RX failsafe channel overrides
        rxfail 3 s 1550
        rxfail 5 s 2000
    #$ OPTION END
#$ OPTION_GROUP END
