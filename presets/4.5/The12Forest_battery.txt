#$ TITLE: The12Forest Battery Cell Voltage Thresholds
#$ FIRMWARE_VERSION: 4.5
#$ CATEGORY: TUNE
#$ STATUS: COMMUNITY
#$ KEYWORDS: battery, voltage, cell, lipo, warning, alarm, The12Forest
#$ AUTHOR: The12Forest
#$ DESCRIPTION: Standard LiPo cell voltage thresholds used by The12Forest across Betaflight 4.5 builds.
#$ DESCRIPTION: These values are per-cell and work correctly regardless of pack cell count (3S, 4S, 6S, etc.)
#$ DESCRIPTION: because Betaflight applies them per-cell automatically when the cell count is detected.
#$ DESCRIPTION: Max cell voltage is 4.20 V (standard LiPo — use 4.35 V for LiHV/HV packs).
#$ DESCRIPTION: Warning threshold is 3.66 V per cell — a conservative early warning to prompt landing.
#$ DESCRIPTION: Minimum cell voltage is 3.50 V per cell — hard cutoff to protect cells from deep discharge.
#$ DESCRIPTION: The optional capacity section sets bat_capacity for mAh-based fuel gauge and cap alarm in OSD.
#$ DESCRIPTION: Only apply the capacity option if your pack matches — this value is build-specific.
#$ DISCUSSION: https://github.com/The12Forest/Betaflight-Presets

defaults nosave


#$ OPTION_GROUP BEGIN: Battery
    #$ OPTION BEGIN (CHECKED): Cell voltage thresholds (4.20 V max / 3.50 V min / 3.66 V warning)
        set vbat_max_cell_voltage = 420
        set vbat_min_cell_voltage = 350
        set vbat_warning_cell_voltage = 366
    #$ OPTION END
    #$ OPTION BEGIN (UNCHECKED): Pack capacity — 1300 mAh (adjust to your pack)
        set bat_capacity = 1300
    #$ OPTION END
#$ OPTION_GROUP END
