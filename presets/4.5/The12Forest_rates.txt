#$ TITLE: The12Forest Rate Profile
#$ FIRMWARE_VERSION: 4.5
#$ CATEGORY: RATES
#$ STATUS: COMMUNITY
#$ KEYWORDS: rates, expo, srate, The12Forest
#$ AUTHOR: The12Forest
#$ DESCRIPTION: Rate profile used by The12Forest across Betaflight 4.5 builds.
#$ DESCRIPTION: Tuned for a relaxed but precise feel — 10% expo on all axes keeps the center region linear
#$ DESCRIPTION: while the 100% super-rate provides plenty of authority at full deflection.
#$ DESCRIPTION: Works well for freestyle, proximity flying, and general cruising where predictable stick response matters more than razor-sharp flips.
#$ DESCRIPTION: Apply to rate profile slot 1. The active profile is restored to slot 3 after saving.
#$ DISCUSSION: https://github.com/The12Forest/Betaflight-Presets

defaults nosave


#$ OPTION_GROUP BEGIN: Rate Profile 1
    #$ OPTION BEGIN (CHECKED): Apply rate profile 1 (10% expo, srate 100)
        rateprofile 1

        set roll_expo = 10
        set pitch_expo = 10
        set yaw_expo = 10
        set roll_srate = 100
        set pitch_srate = 100
        set yaw_srate = 100
    #$ OPTION END
#$ OPTION_GROUP END


# restore original rateprofile selection
rateprofile 3
