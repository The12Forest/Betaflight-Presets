#$ KEYWORDS: gyro, filter, lpf, dynamic, simplified, The12Forest
#$ AUTHOR: The12Forest
#$ DESCRIPTION: Gyro and filter configuration used by The12Forest on Betaflight 4.5 builds.
#$ DESCRIPTION: LPF1 static is disabled (0 Hz) — the dynamic notch filter handles low-frequency noise instead,
#$ DESCRIPTION: avoiding the phase lag that a static LPF1 would introduce on a clean build.
#$ DESCRIPTION: LPF2 static is set to 1000 Hz as a high-frequency alias rejection stage above the gyro Nyquist.
#$ DESCRIPTION: Dynamic LPF1 min is set to 0 Hz so the filter can track all the way down if needed.
#$ DESCRIPTION: Simplified gyro filter multiplier at 200% raises the dynamic filter cutoffs — suitable for
#$ DESCRIPTION: a well-balanced 5-inch build with low vibration. Lower this value on noisier or larger builds.
#$ DISCUSSION: https://github.com/The12Forest/Betaflight-Presets

defaults nosave


#$ OPTION_GROUP BEGIN: Gyro & Filters
    #$ OPTION BEGIN (CHECKED): Apply gyro / filter settings
        set gyro_lpf1_static_hz = 0
        set gyro_lpf2_static_hz = 1000
        set gyro_lpf1_dyn_min_hz = 0
        set simplified_gyro_filter_multiplier = 200
    #$ OPTION END
#$ OPTION_GROUP END
